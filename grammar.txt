STMT = | "OUTPUT" EXPR
       | "INPUT" NAME
       | "NEW" NAME
       | "PUT" NAME EXPR
       | "IF" EXPR STMT* ("ELSE" STMT*)? "END"
       | "WHILE" EXPR STMT* "END"

EXPR = | STRING
       | INT
       | FLOAT
       | "VARIABLE" NAME
       | "ADD" EXPR EXPR
       | "SUBTRACT" EXPR EXPR
       | "MULTIPLY" EXPR EXPR
       | "DIVIDE" EXPR EXPR
       | "EQUAL" EXPR EXPR
       | "GREATER" EXPR EXPR
       | "LESS" EXPR EXPR
